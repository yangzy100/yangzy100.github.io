Venn Diagram
Create Venn Diagram with various R packages.
2021-09-22 00:00 folder [R]   label plot  

Venn Diagram can be created by various Packages: ggven, ggVennDiagram, RAM, and VennDiagram

Generate Data

# Generate 3 sets of 200 words
set1 <- paste(rep("word_" , 200) , sample(c(1:1000) , 200 , replace=F) , sep="")
set2 <- paste(rep("word_" , 200) , sample(c(1:1000) , 200 , replace=F) , sep="")
set3 <- paste(rep("word_" , 200) , sample(c(1:1000) , 200 , replace=F) , sep="")

# List of items
x <- list(A=set1, B=set2)

# List of items
y <- list(A=set1, B=set2, C=set3)
names(y)
## [1] "A" "B" "C"

1) Create Venn diagrams using the ggven R package

2D Venn diagram

library("ggvenn")

ggvenn(y, columns = c("A", "B"),stroke_size = 0.5)

3D Venn diagram

2) Using the ggVennDiagram

library(ggVennDiagram)
# 2D Venn diagram
ggVennDiagram(x)

# 3D Venn diagram
ggVennDiagram(y)

Customize border and category names of the Venn diagram

library(ggplot2)

ggVennDiagram(y, category.names = c("Group 1","Group 2","Group 3"), 
              color = "black", lwd = 0.8, lty = 1, label_alpha = 0) + 
  scale_fill_gradient(low = "yellow", high = "lightblue")

3) Using the RAM package

library(RAM)
group.venn(list(Group1=set1, Group2=set2), label=F, 
           fill = c("orange", "blue"),
           cat.pos = c(0, 0),
           lab.cex=1.1)

4) Using the VennDiagram Package

library(VennDiagram)

v<-venn.diagram(
  x = list(set1, set2, set3),
  category.names = c("Set 1" , "Set 2 " , "Set 3"),
  fill = c("orange", "blue","pink"),
  alpha = c(0.5, 0.5, 0.5), cat.cex = 1.5, cex=1.5,
  filename=NULL
)

# have a look at the default plot
grid.newpage()
grid.draw(v)

Customize it

# Prepare a palette of 3 colors with R colorbrewer:
library(RColorBrewer)
myCol <- brewer.pal(3, "Pastel2")

# Chart
v2<-venn.diagram(
  x = list(set1, set2, set3),
  category.names = c("Set 1" , "Set 2 " , "Set 3"),
  alpha = c(0.5, 0.5, 0.5), 
  filename=NULL,

  # Circles
  lwd = 2,
  lty = 'blank',
  fill = myCol,
  
  # Numbers
  cex = .6,
  fontface = "bold",
  fontfamily = "sans",
  
  # Set names
  cat.cex = 1.2,
  cat.fontface = "bold",
  cat.default.pos = "outer",
  cat.fontfamily = "sans",
  rotation = 1
)

v2[[11]]$label<-paste(paste0('Intersection = ',length(intersect(intersect(set1,set2),set3)),"\n",
                      paste(intersect(intersect(set1,set2),set3), collapse="\n"))) 

# plot  
grid.newpage()
grid.draw(v2)
